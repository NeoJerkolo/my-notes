Порівняння — у чому різниця?

Чотири ключові концепції

Iterable
Об'єкт з __iter__. Можна обійти for-циклом. Сам НЕ зберігає позицію.
Iterator

Generator

Gen expression
Коротка форма генератора. Синтаксис як list comprehension, але в дужках ().

Ітератор (Iterator)

Ітераторний протокол

Два обов'язкових методи. Без них — не ітератор.

class Counter:
    """Власний ітератор від 1 до max"""
    def __init__(self, max):
        self.max = max
        self.current = 0

    def __iter__(self):
        # Повертає self — ітератор сам є iterable!
        return self

    def __next__(self):
        self.current += 1
        if self.current > self.max:
            raise StopIteration  # сигнал кінця
        return self.current

c = Counter(3)
next(c)  # → 1
next(c)  # → 2
for n in Counter(5):
    print(n)  # 1, 2, 3, 4, 5

Як працює for під капотом

# Цей код:
for x in [1, 2, 3]:
    print(x)

# Рівнозначний цьому:
_iter = iter([1, 2, 3])  # → list.__iter__()
while True:
    try:
        x = next(_iter)    # → list_iterator.__next__()
        print(x)
    except StopIteration:
        break              # for ловить StopIteration

# iter() викликає __iter__
# next() викликає __next__
# StopIteration → кінець циклу
Починаючи з Python 3.7 StopIteration всередині генератора перетворюється на RuntimeError. Завжди використовуй return замість raise StopIteration у генераторах.

Ітератор — одноразовий!

from collections.abc import Iterator

lst = [1, 2, 3]
it = iter(lst)

# Перший обхід
list(it)  # → [1, 2, 3]

# Другий обхід — ПОРОЖНЬО!
list(it)  # → []  ← ітератор вичерпано

# Список — не ітератор, можна обходити знову:
list(lst)  # → [1, 2, 3]  ✓
list(lst)  # → [1, 2, 3]  ✓

# iter() на ітераторі повертає той самий об'єкт!
iter(it) is it  # → True
Ніколи не збережи ітератор і не чекай повторного обходу. Якщо потрібен повторний обхід — зроби list(it) або використай iterable.

Практичний ітератор — зчитування файлу

class FileLineIterator:
    """Читає файл рядок за рядком — O(1) пам'яті"""
    def __init__(self, path):
        self._file = open(path, 'r')

    def __iter__(self):
        return self

    def __next__(self):
        line = self._file.readline()
        if not line:
            self._file.close()
            raise StopIteration
        return line.rstrip('\n')

# Використання:
for line in FileLineIterator('data.txt'):
    print(line)

Ітерований об'єкт (Iterable)

Iterable vs Iterator — ключова різниця

class NumberRange:
    """Iterable — можна обходити багато разів!"""
    def __init__(self, start, stop):
        self.start = start
        self.stop  = stop

    def __iter__(self):
        # Щоразу створює НОВИЙ ітератор
        current = self.start
        while current < self.stop:
            yield current         # ← це генератор!
            current += 1

r = NumberRange(1, 4)
list(r)  # → [1, 2, 3]
list(r)  # → [1, 2, 3]  ← знову! Не вичерпано.

# iter(r) is iter(r) → False (новий кожного разу)
Ключова різниця: __iter__ повертає self (ітератор) або новий об'єкт (iterable). Якщо повертає self — об'єкт є ітератором і вичерпується.

__getitem__ — альтернативний протокол

class OldStyleIterable:
    """Без __iter__! Але Python вміє ітерувати."""
    def __init__(self, data):
        self.data = data

    def __getitem__(self, index):
        return self.data[index]  # IndexError → stop

obj = OldStyleIterable([10, 20, 30])
for x in obj:
    print(x)  # 10, 20, 30

# Python робить iter() → шукає __iter__,
# якщо нема → шукає __getitem__(0, 1, 2...)
# поки не отримає IndexError
Це legacy-протокол. У новому коді завжди реалізуй __iter__.

Генератор (Generator function)

Основи генераторів

Функція стає генератором як тільки в ній є хоча б одне yield.

def simple_gen():
    print("крок 1")
    yield 1          # ← зупинка, повертає 1
    print("крок 2")
    yield 2          # ← зупинка, повертає 2
    print("крок 3")
    yield 3

g = simple_gen()  # нічого не виконалось!
next(g)  # → "крок 1" → 1
next(g)  # → "крок 2" → 2
next(g)  # → "крок 3" → 3
next(g)  # → StopIteration!
Генераторна функція при виклику повертає об'єкт-генератор, але не виконує жодного коду! Код запускається лише при next().

yield — пауза та передача значення

def infinite_counter(start=0):
    """Нескінченна послідовність — О(1) пам'яті"""
    n = start
    while True:
        yield n
        n += 1

import itertools
for i, val in zip(
    range(5),
    infinite_counter(10)
):
    print(val)  # 10, 11, 12, 13, 14

# yield from — делегування
def chain_gen(*iterables):
    for it in iterables:
        yield from it   # як for x in it: yield x

list(chain_gen([1,2], [3,4]))  # [1, 2, 3, 4]

return у генераторі

def gen_with_return():
    yield 1
    yield 2
    return "готово"  # стає value StopIteration

g = gen_with_return()
next(g)  # → 1
next(g)  # → 2
try:
    next(g)
except StopIteration as e:
    print(e.value)  # → "готово"

# yield from захоплює return value:
def delegator():
    result = yield from gen_with_return()
    print(f"результат: {result}")  # "готово"

send(), throw(), close() — двостороннє спілкування

def accumulator():
    total = 0
    while True:
        value = yield total   # ← yield повертає sent
        if value is None:
            break
        total += value

g = accumulator()
next(g)       # → 0  (треба "прокрутити" до yield)
g.send(10)   # → 10
g.send(5)    # → 15
g.send(3)    # → 18

# throw() — кинути виключення в генератор
def safe_gen():
    try:
        yield 1
    except ValueError:
        print("Перехоплено!")
        yield -1

g = safe_gen()
next(g)               # → 1
g.throw(ValueError)   # → "Перехоплено!" → -1

# close() — завершити генератор (GeneratorExit)
g.close()

Ланцюгування генераторів — pipeline

def read_lines(path):
    with open(path) as f:
        yield from f

def strip_comments(lines):
    for line in lines:
        if not line.startswith('#'):
            yield line.strip()

def parse_ints(lines):
    for line in lines:
        if line.isdigit():
            yield int(line)

# Весь pipeline — O(1) пам'яті!
lines   = read_lines('data.txt')
clean   = strip_comments(lines)
numbers = parse_ints(clean)

total = sum(numbers)  # обробляє по 1 рядку
Генераторні pipeline — ідеальний патерн для ETL, обробки логів, потокових даних. Жодного рядка в пам'яті до обробки.

Генератор зі станом — coroutine-стиль

def running_average():
    """Обчислює середнє наростаючим підсумком"""
    total = 0.0
    count = 0
    avg   = None
    while True:
        value = yield avg
        if value is None:
            return
        total += value
        count += 1
        avg = total / count

ra = running_average()
next(ra)       # ініціалізація → None
ra.send(10)   # → 10.0
ra.send(20)   # → 15.0
ra.send(30)   # → 20.0

Рекурсивний генератор — обхід дерева

def flatten(nested):
    """Розгорнути довільно вкладену структуру"""
    for item in nested:
        if isinstance(item, (list, tuple)):
            yield from flatten(item)
        else:
            yield item

data = [1, [2, [3, 4], 5], [6, [7]]]
list(flatten(data))  # [1, 2, 3, 4, 5, 6, 7]

# Обхід дерева (BFS)
def bfs(root):
    queue = [root]
    while queue:
        node = queue.pop(0)
        yield node.value
        queue.extend(node.children)

Контекстні менеджери (Context Managers)

Протокол контекстного менеджера

Два методи: __enter__ (setup) та __exit__ (teardown). Гарантований cleanup навіть при виключенні.

class Timer:
    import time
    def __enter__(self):
        self.start = __import__('time').time()
        return self   # ← значення в "as"

    def __exit__(self, exc_type, exc_val, tb):
        self.elapsed = (
            __import__('time').time() - self.start
        )
        print(f"Пройшло: {self.elapsed:.3f}с")
        return False  # False = не пригнічувати
                      # виключення

with Timer() as t:
    do_heavy_work()
# Пройшло: 1.234с
# t.elapsed → 1.234

__exit__ — обробка виключень

class Suppress:
    """Пригнічує вказані виключення"""
    def __init__(self, *exceptions):
        self.exceptions = exceptions

    def __enter__(self):
        return self

    def __exit__(self, exc_type, exc_val, tb):
        # True → пригнітити виключення
        # False/None → пропустити далі
        if exc_type and issubclass(
            exc_type, self.exceptions
        ):
            self.exception = exc_val
            return True   # поглинаємо!
        return False

with Suppress(FileNotFoundError):
    data = open("maybe.txt").read()
# Продовжує без помилки

# Є в stdlib: contextlib.suppress
from contextlib import suppress
with suppress(FileNotFoundError): ...

@contextmanager — через генератор

from contextlib import contextmanager

@contextmanager
def managed_db(url):
    conn = connect(url)      # __enter__
    try:
        yield conn           # ← тут виконується with
    except Exception:
        conn.rollback()      # обробка помилки
        raise
    finally:
        conn.close()         # __exit__ завжди

with managed_db("sqlite:///test.db") as db:
    db.execute("INSERT INTO ...")
    # При будь-якій помилці — rollback + close
@contextmanager — найчистіший спосіб. yield повинен бути рівно один. finally гарантує cleanup.

# Вкладені with — два стилі
# Старий:
with open('a') as a:
    with open('b') as b:
        ...

# Новий (Python 3.1+):
with open('a') as a, open('b') as b:
    ...

# Python 3.10+ — дужки для кількох рядків:
with (
    open('a') as a,
    open('b') as b,
    open('c') as c,
):
    ...

# Повторне використання (reentrant)
from contextlib import redirect_stdout
import io
cm = redirect_stdout(io.StringIO())
with cm: with cm: ...  # ок для reentrant CM

Патерн: декоратор + контекстний менеджер

@contextmanager
def transaction(session):
    try:
        yield session
        session.commit()
    except:
        session.rollback()
        raise

# Використання і як CM і як декоратор
@transaction(session)   # ← як декоратор!
def create_user(name):
    session.add(User(name=name))

with transaction(session) as s:  # ← як CM
    s.add(User(name="Alice"))
contextmanager декоратор автоматично підтримує і with і @ використання — один код для двох патернів.
