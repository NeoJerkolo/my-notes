pytest Guru Cheatsheet

Python 3.9 · pytest 7.x — від основ до архітектурного рівня

Основи pytest

Структура тесту

Кожна функція, що починається з test_, або клас з Test — тест.

# test_example.py
def test_addition():
    result = 2 + 2
    assert result == 4

def test_string():
    s = "hello"
    assert s.upper() == "HELLO"
    assert "ell" in s

# Клас-тест (без __init__!)
class TestMath:
    def test_multiply(self):
        assert 3 * 4 == 12

    def test_divide(self):
        assert 10 / 2 == 5.0

Assert — детальний вивід

pytest переписує assert для детальних повідомлень про помилки — не потрібен unittest.

# pytest показує різницю автоматично
def test_list():
    expected = [1, 2, 3]
    actual   = [1, 2, 4]
    assert actual == expected
    # AssertionError:
    #   assert [1, 2, 4] == [1, 2, 3]
    #   At index 2 diff: 4 != 3

# Власне повідомлення
assert result > 0, f"Очікували > 0, отримали {result}"

# Порівняння рядків — diff по рядках
assert got_text == expected_text
pytest використовує AST-rewriting — assert перетворюється у момент збору тестів, а не під час запуску.

Фікстури (Fixtures)

Базові фікстури

Фікстури — функції, що забезпечують тести даними або ресурсами. Передаються як аргументи.

import pytest

@pytest.fixture
def user():
    return {"name": "Alice", "age": 30}

@pytest.fixture
def admin_user(user):  # фікстура у фікстурі!
    user["role"] = "admin"
    return user

def test_user_name(user):
    assert user["name"] == "Alice"

def test_admin(admin_user):
    assert admin_user["role"] == "admin"

Scope — область видимості

Контролює, як часто фікстура створюється.

@pytest.fixture(scope="function")  # за замовч.
def func_fix(): ...  # новий для кожного тесту

@pytest.fixture(scope="class")
def class_fix(): ...  # одна на клас

@pytest.fixture(scope="module")
def db_connection():  # одна на файл
    conn = create_connection()
    yield conn          # teardown після yield
    conn.close()

@pytest.fixture(scope="session")
def app_server():    # одна на весь запуск
    server = start_server()
    yield server
    server.stop()
Фікстура з нижчим scope не може залежати від вищого! function не може взяти session фікстуру.

yield — setup та teardown

yield розділяє setup (до) і teardown (після). Гарантує очищення навіть при помилці.

@pytest.fixture
def temp_file(tmp_path):
    f = tmp_path / "data.txt"
    f.write_text("hello")
    yield f              # ← тест отримує це
    # teardown — виконується після тесту
    f.unlink(missing_ok=True)

@pytest.fixture
def database():
    db = Database.connect("test.db")
    db.migrate()
    yield db
    db.rollback()        # відкат після кожного тесту
    db.disconnect()
tmp_path — вбудована фікстура pytest. Надає тимчасову директорію як pathlib.Path.

autouse — автоматична фікстура

@pytest.fixture(autouse=True)
def reset_env(monkeypatch):
    # виконується для КОЖНОГО тесту в модулі
    monkeypatch.setenv("ENV", "test")

@pytest.fixture(autouse=True, scope="session")
def setup_logging():
    import logging
    logging.basicConfig(level=logging.DEBUG)

# У conftest.py autouse=True → діє на весь пакет
Не зловживай autouse — складніше відстежити, що відбувається у тесті.

params у фікстурах

Фікстура з params автоматично запускає тест для кожного значення.

@pytest.fixture(params=["sqlite", "postgres"])
def db_backend(request):
    backend = request.param   # поточне значення
    db = Database(backend)
    yield db
    db.close()

def test_query(db_backend):
    # запуститься 2 рази: для sqlite і postgres
    result = db_backend.query("SELECT 1")
    assert result is not None

request — метаінформація фікстури

@pytest.fixture
def smart_fixture(request):
    # Доступ до мета-інформації тесту
    test_name = request.node.name
    module = request.module
    cls = request.cls       # None якщо не в класі
    scope = request.scope
    param = request.param   # якщо є params

    # Фіналізатор (альтернатива yield)
    def cleanup():
        print(f"Очищення після {test_name}")
    request.addfinalizer(cleanup)

    return {"test": test_name}

Маркери (Markers)

Вбудовані маркери

import pytest
import sys

# Пропустити тест
@pytest.mark.skip(reason="ще не реалізовано")
def test_feature(): ...

# Очікуємо падіння (xfail)
@pytest.mark.xfail(reason="відомий баг #123")
def test_buggy():
    assert 1 == 2  # XFAIL — очікувана помилка

# xfail strict — якщо раптом пройде, це помилка!
@pytest.mark.xfail(strict=True, reason="має падати")
def test_strict_xfail(): ...

# Умовний skip
@pytest.mark.skipif(
    sys.platform == "win32",
    reason="не підтримується на Windows"
)
def test_unix_only(): ...

Власні маркери

# pytest.ini — реєстрація маркерів
[pytest]
markers =

import pytest

@pytest.mark.parametrize("input, expected", [

    (3, 9),

    mock_db.query.assert_called_with("SELECT *")

@patch("myapp.services.requests.get")

Виключення та попередження

pytest.raises

import pytest

def test_raises_basic():
    with pytest.raises(ValueError):
        int("не число")

# Перевірка повідомлення
def test_raises_message():
    with pytest.raises(ValueError, match=r"invalid"):
        raise ValueError("invalid literal")

# Детальна перевірка ExceptionInfo
def test_exception_info():
    with pytest.raises(TypeError) as exc_info:
        "text" + 5
    
    exc = exc_info.value
    assert "str" in str(exc)
    assert exc_info.type is TypeError

# Ланцюгові виключення
def test_chained():
    with pytest.raises(RuntimeError) as exc:
        try:
            1/0
        except ZeroDivisionError as e:
            raise RuntimeError("wrap") from e
    assert exc.value.__cause__ is not None

capsys — захоплення stdout/stderr

def test_print_output(capsys):
    print("Hello, World!")
    print("Error!", file=__import__('sys').stderr)
    
    captured = capsys.readouterr()
    assert "Hello" in captured.out
    assert "Error" in captured.err

# caplog — захоплення логів
import logging

def test_logging(caplog):
    import logging
    with caplog.at_level(logging.WARNING):
        logging.warning("Попередження!")
        logging.info("Info (не захопить)")
    
    assert "Попередження" in caplog.text
    assert caplog.records[0].levelname == "WARNING"

Property-based testing (Hypothesis)

# pip install hypothesis
from hypothesis import given, strategies as st

@given(st.integers(), st.integers())
def test_add_commutative(a, b):
    # Hypothesis генерує сотні кейсів!
    assert add(a, b) == add(b, a)

@given(st.lists(st.integers(), min_size=1))
def test_sort_idempotent(lst):
    assert sorted(sorted(lst)) == sorted(lst)

@given(st.text(min_size=1))
def test_encode_decode(s):
    assert decode(encode(s)) == s

Власний Assert Helper

# Кастомний assert для складних об'єктів
class ResponseAssertion:
    def __init__(self, response):
        self.response = response

    def has_status(self, code):
        assert self.response.status_code == code, (
            f"Очікували {code}, отримали "
            f"{self.response.status_code}: "
            f"{self.response.text}"
        )
        return self  # ← ланцюгові виклики

    def has_json_key(self, key):
        data = self.response.json()
        assert key in data, f"Ключ {key!r} відсутній"
        return self

def assert_response(r): return ResponseAssertion(r)

# Використання:
assert_response(r).has_status(200).has_json_key("id")

Розбір виводу та коди виходу

# Exit codes:
# 0 — всі тести пройшли
# 1 — є падаючі тести
# 2 — перервано (Ctrl+C)
# 3 — внутрішня помилка pytest
# 4 — помилка в команді
# 5 — жодного тесту не знайдено

# Символи у виводі:
# .  — PASSED
# F  — FAILED
# E  — ERROR (в фікстурі)
# x  — XFAIL (очікувана помилка)
# X  — XPASS (несподівано пройшло)
# s  — SKIPPED

# -ra: показати підсумок всіх != pass
pytest -ra
# -rN: failed; -rs: skipped; -rx: xfail
